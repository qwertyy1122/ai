

# Tower of Hanoi using Hill Climbing

import copy

# Initial state (all disks on rod A)
def initial_state(n):
    return [list(range(n, 0, -1)), [], []]

# Goal check
def is_goal(state, n):
    return state[2] == list(range(n, 0, -1))

# Heuristic: number of disks in correct position (rod C)
def heuristic(state, n):
    return len(state[2])

# Generate all valid moves
def generate_neighbors(state):
    neighbors = []

    for i in range(3):  # from rod
        if not state[i]:
            continue

        for j in range(3):  # to rod
            if i != j:
                if not state[j] or state[i][-1] < state[j][-1]:
                    new_state = copy.deepcopy(state)
                    disk = new_state[i].pop()
                    new_state[j].append(disk)
                    neighbors.append(new_state)

    return neighbors

# Hill Climbing Algorithm
def hill_climbing(n):
    current = initial_state(n)
    steps = 0

    print("Initial State:", current)

    while True:
        if is_goal(current, n):
            print("\nGoal Reached!")
            return current

        current_h = heuristic(current, n)
        neighbors = generate_neighbors(current)

        best_neighbor = None
        best_h = current_h

        for neighbor in neighbors:
            h = heuristic(neighbor, n)
            if h > best_h:
                best_neighbor = neighbor
                best_h = h

        if best_neighbor is None:
            print("\nStuck at Local Optimum!")
            return current

        current = best_neighbor
        steps += 1

        print(f"Step {steps}: {current} | h = {best_h}")

# ---------------- Main ----------------
if __name__ == "__main__":
    n = 3  # number of disks
    final_state = hill_climbing(n)
